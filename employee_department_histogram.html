import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns

# Load dataset
df = pd.read_csv("employees.csv")

# Frequency count for Marketing department
marketing_count = (df["Department"] == "Marketing").sum()
print("Verification email: 23f2003677@ds.study.iitm.ac.in")
print("Frequency count for 'Marketing' department:", marketing_count)

# Plot histogram of department counts
plt.figure(figsize=(6,4))
sns.countplot(x="Department", data=df, order=df["Department"].value_counts().index)
plt.xticks(rotation=45)
plt.title("Employee Count by Department")
plt.tight_layout()
plt.savefig("employee_department_histogram.png")
plt.show()
